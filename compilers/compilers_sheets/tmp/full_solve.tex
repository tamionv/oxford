\newcommand{\Proof}[2]{

}

    \newlength{\bcaselen}
    \settowidth{\bcaselen}{\textit{BC:....}}
    \begin{minipage}[t]{\linewidth-\bcaselen}
    #1
    \end{minipage}

    \leavevmode
    \\

    \textit{IS:}
    \newlength{\indsteplen}
    \settowidth{\indsteplen}{\textit{IS:....}}
    \begin{minipage}[t]{\linewidth-\indsteplen}
    #2
    \end{minipage}

    \leavevmode
    \\
}
